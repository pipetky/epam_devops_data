2. modify option vm.dirty_ratio:
   - using echo utility
   - using sysctl utility
   - using sysctl configuration files

```
[root@localhost superadmin]# cat /proc/sys/vm/dirty_ratio
30

[root@localhost superadmin]# echo 40 > /proc/sys/vm/dirty_ratio

40

[root@localhost superadmin]# cat /proc/sys/vm/dirty_ratio
```
